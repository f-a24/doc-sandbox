Statefull Component

```jsx
class App extend React.Component {
  constructor(props) {
    super(props);
    this.state = { data: '' };
    this.inputHandler = this.inputHandler.bind(this);
  }
  inputHandler (e) {
    this.setState({ data: e.target.value });
  }
  render() {
    return (
      <div className="msg">
        <p style={{color: 'skyblue'}}>{this.state.data}</p>
        <Input onChange={this.inputHandler} />
      </div>
    );
  }
}
```
